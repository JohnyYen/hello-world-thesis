	\begingroup
	\begin{table}[H]
		\centering
		\caption{#1} % Parámetro 1: El título de la tabla
		\label{#2}   % Parámetro 2: El identificador para referenciar (\ref{})
		\vspace{6pt}
		\renewcommand{\arraystretch}{1.4} 
		
		\begin{tabularx}{\textwidth}{ | >{\bfseries}p{4.5cm} | X | }
			\hline
			#3 % Parámetro 3: Todo el contenido interno de la tabla
			\hline
		\end{tabularx}
	\end{table}
	\endgroup
}


% Función para generar tablas de Descripción de Casos de Uso
% Parámetro 1: Nombre del Caso de Uso (para el título)
% Parámetro 2: Label para referenciar (\ref{})
% Parámetro 3: Contenido de la tabla
\newcommand{\usecase}[3]{
	\begingroup
	\begin{table}[H]
		\centering
		\caption{#1} 
		\label{#2}   
		\renewcommand{\arraystretch}{1.3} 
		
		% m{3.5cm} -> Ancho fijo para los nombres de los campos
		% X -> El resto del ancho de la página para la descripción
		% >{\bfseries} -> Pone en negrita automático la primera columna
		\begin{tabularx}{\textwidth}{ | >{\bfseries}m{3.5cm} | X | }
			\hline
			#3 
			\hline
		\end{tabularx}
	\end{table}
	\endgroup
}

\NewDocumentCommand{\testCaseTable}{m m m m +m}{%
	\noindent
	\adjustbox{max width=\textwidth}{%
		\begin{tblr}{
				width   = \textwidth,
				colspec = {Q[c,wd=0.6cm,m] Q[l,wd=2.8cm,m] X[l,m] X[l,m] X[l,m] X[l,m]},
				hlines, vlines,
				rowsep  = 4pt,
				leftsep = 3pt,
				rightsep= 3pt,
				cells   = {valign=m, font=\small},
			}
			\SetCell[c=6]{l} \textbf{Caso de Prueba} \\
			\SetCell[c=2]{l} \textbf{Nombre} & & \SetCell[c=4]{l} #1 & & & \\
			\SetCell[c=2]{l} \textbf{Objetivo de la Prueba} & & \SetCell[c=4]{l} #2 & & & \\
			\SetCell[c=2]{l} \textbf{Descripción de la prueba} & & \SetCell[c=4]{l} #3 & & & \\
			\SetCell[c=2]{l} \textbf{Condiciones de entrada} & & \SetCell[c=4]{l} #4 & & & \\
			\SetCell[c=6]{l} \textbf{Combinaciones de valores} \\
			\textbf{CP} & \textbf{Escenarios} & \textbf{Variables} & \textbf{Valores de entrada} & \textbf{Resultado esperado} & \textbf{Resultado real} \\
			#5
		\end{tblr}
	}%
}

\ExplSyntaxOn

\newcounter{reqcounter}
\setcounter{reqcounter}{0}
\renewcommand{\thereqcounter}{RF-\arabic{reqcounter}}
\renewcommand\theadfont{\bfseries}

\tl_new:N \g_req_rows_tl

% #1 = título (también usado en caption)
% #2 = label opcional para la tabla (tab:...)
% #3 = lista de requisitos separados por ;
\NewDocumentCommand{\requisitos}{ m o m }
{
	\subsubsection*{#1}
	
	% Limpiar acumulador
	\tl_gclear:N \g_req_rows_tl
	
	% Partir la lista y acumular filas
	\seq_set_split:Nnn \l_tmpa_seq { ; } { #3 }
	\seq_map_inline:Nn \l_tmpa_seq
	{
		\tl_set:Nx \l_tmpb_tl { \tl_trim_spaces:n { ##1 } }
		\tl_if_empty:NF \l_tmpb_tl
		{
			\tl_gput_right:Nx \g_req_rows_tl
			{
				\exp_not:n { \hline \refstepcounter{reqcounter} }
				\exp_not:n { \thereqcounter }
				\exp_not:n { \label{rf:\arabic{reqcounter}} }
				\exp_not:n { & }
				\exp_not:V \l_tmpb_tl
				\exp_not:n { \\ }
			}
		}
	}
	
	% Volcar dentro de la tabla
	\begin{longtable}{|p{0.12\linewidth}|p{0.82\linewidth}|}
		\caption{#1}
		\IfValueT{#2}{ \label{#2} }  % ← Aplicamos el label AQUÍ, dentro del longtable
		\\
		
		\hline
		\textbf{ID} & \thead[l]{Descripción} \\
		\hline
		\endfirsthead
		\multicolumn{2}{c}{\tablename\ -- continuación} \\
		\hline
		\textbf{ID} & \thead[l]{Descripción} \\
		\hline
		\endhead
		\hline
		\endfoot
		\hline
		\endlastfoot
		\g_req_rows_tl
		\hline
	\end{longtable}
}
